Atlas

A procedural generation toolkit for Roblox: seeded noise with analytical derivatives (Noise), point scattering (Sampling), response curves (Curve), and on-screen debugging (Visualizer). 

local Atlas = require(path.to.Atlas)

local simplex = Atlas.Noise.Simplex.new(42)
local falloff = Atlas.Curve.new({
	Vector2.new(0, 0),
	Vector2.new(0.6, 0.15),
	Vector2.new(1, 1),
})

local height = falloff:evaluate(Atlas.Noise.fbm2D(simplex, x * 0.01, z * 0.01, 4) * 0.5 + 0.5)
